gcd

用(a, b)表示gcd(a, b)

(a, b) = (b, a % b)

(a, b) = (b, a - b) (a > b)

exgcd

求解ax + by = gcd(a, b)


void exgcd(int a, int b, int &g, int &x, int &y)
{
	if (!b) { x = 1; y = 0; g = a; return; }
	exgcd(b, a % b, g, y, x), y -= (a / b) * x;
}

int CRT()
{
    int ans = 0, M = 1, x, y;
    for (int i = 1; i <= k; i++) M *= m[i];
    for (int i = 1; i <= k; i++)
    {
        int tp = M / m[i];
        exgcd(tp,b[i],x,y);
        x = get_inv(tp, m[i])
        ans = (ans + tp * x * a[i]) % M;
    }
    return (ans + M) % M;
}

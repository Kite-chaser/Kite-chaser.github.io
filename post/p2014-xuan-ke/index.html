#include<iostream>
#include<cstdio>
#include<cstring>
#define N 400
using namespace std;
int n,m;
int k[N],s[N];
int d[N][N];
int sz[N];
int first[N],nxt[N],to[N],len;
void add(int u,int v){
   to[++len]=v;
   nxt[len]=first[u];
   first[u]=len;
}
void dfs(int u){
   sz[u]=1;
   for(int i=first[u];i!=-1;i=nxt[i]){
   	dfs(to[i]);
   	sz[u]+=sz[to[i]];
   	for(int j=min(m,sz[u]);j>=0;j--){
   		for(int k=min(j-1,sz[to[i]]);k>=0;k--) d[u][j]=max(d[u][j],d[u][j-k-1]+d[to[i]][k]+s[to[i]]);
   	}
   }
}
int main()
{
   memset(first,-1,sizeof(first));
   scanf("%d%d",&n,&m);
   for(int i=1;i<=n;i++){
   	scanf("%d%d",&k[i],&s[i]);
   	add(k[i],i);
   }
   dfs(0);
   printf("%d",d[0][m]);
   return 0;
}

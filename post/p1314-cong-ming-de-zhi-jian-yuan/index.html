#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#define N 200010
#define ll long long
using namespace std;
ll n,m,k;
ll W;
ll ans=999999999999999999,num;
ll l=1,r,mid;
ll s[N],t[N];
ll sum1[N],sum2[N];
ll w[N],v[N];
bool sl(ll cnt){
	ll num=0;
	memset(sum1,0,sizeof(sum1));
	memset(sum2,0,sizeof(sum2));
	for(ll i=1;i<=n;i++){
		if(w[i]>=cnt) sum1[i]=sum1[i-1]+1,sum2[i]=sum2[i-1]+v[i];
		else sum1[i]=sum1[i-1],sum2[i]=sum2[i-1];
	}
	for(ll i=1;i<=m;i++){
		num+=(sum1[t[i]]-sum1[s[i]-1])*(sum2[t[i]]-sum2[s[i]-1]);
	}
	if(num>k){
		ans=min(ans,num-k);
		return false;
	}
	ans=min(ans,k-num);
	return true;
}
int main()
{
	scanf("%lld%lld%lld",&n,&m,&k);
	for(ll i=1;i<=n;i++) scanf("%lld%lld",&w[i],&v[i]),r=max(r,w[i]);
	for(ll i=1;i<=m;i++) scanf("%lld%lld",&s[i],&t[i]);
	r+=2;
	while(l<r){
		mid=(l+r)/2;
		if(sl(mid)) r=mid;
		else l=mid+1;
	}
	printf("%lld",ans);
	return 0;
}
